          <div class="tx-name">${t.name}</div>
          <div class="tx-date">${t.date}</div>
        </div>
        <div class="tx-amount ${t.type}">${t.amount > 0 ? '+' : ''}${t.amount.toFixed(2)} €</div>
      </div>
    `).join('');
  }

  function login() {
    document.getElementById('login-screen').style.display = 'none';
    document.getElementById('dashboard').style.display = 'block';

  }

  function logout() {
    document.getElementById('dashboard').style.display = 'none';
    document.getElementById('login-screen').style.display = 'flex';
  }

  function init() {
    const d = new Date();
    document.getElementById('current-date').textContent = d.toLocaleDateString('fr-FR', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });
    renderTx(document.getElementById('tx-list'), transactions.slice(0, 5));
    renderTx(document.getElementById('tx-list-full'), transactions);

  function showPage(name, el) {
    document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
    document.getElementById('page-' + name).classList.add('active');
    document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));

  function openModal(name) { document.getElementById('modal-' + name).classList.add('open'); }
  function closeModal(name) { document.getElementById('modal-' + name).classList.remove('open'); }

    t.textContent = msg;

  function updateSolde() {
    document.getElementById('solde-display').textContent = solde.toLocaleString('fr-FR', { minimumFractionDigits: 2 }) + ' €';
  }

  function confirmVirement() {
    const ben = document.getElementById('mv-ben').value;
    const amt = parseFloat(document.getElementById('mv-amount').value);
    if (!ben || !amt || amt <= 0) { showToast('❌ Veuillez remplir tous les champs'); return; }
    if (amt > solde) { showToast('❌ Solde insuffisant'); return; }
    solde -= amt;
    updateSolde();
    transactions.unshift({ icon: '↗️', type: 'debit', name: `Virement — ${ben}`, date: "Aujourd'hui", amount: -amt, cat: 'debit' });
    renderTx(document.getElementById('tx-list'), transactions.slice(0, 5));
    closeModal('virement');
    showToast(`✅ Virement de ${amt} € envoyé à ${ben}`);
  }

    if (!amt || amt <= 0) { showToast('❌ Montant invalide'); return; }
    solde += amt;
    updateSolde();
    transactions.unshift({ icon: '💵', type: 'credit', name: 'Dépôt simulé', date: "Aujourd'hui", amount: amt, cat: 'credit' });
    renderTx(document.getElementById('tx-list'), transactions.slice(0, 5));
    closeModal('depot');
    showToast(`✅ Dépôt de ${amt} € effectué`);
  }

  function confirmBloquer() {
    closeModal('bloquer');
    showToast('🔒 Carte bloquée temporairement');
  }

  function doVirement() {
    const ben = document.getElementById('v-beneficiaire').value;
    const amt = parseFloat(document.getElementById('v-montant').value);
    const ref = document.getElementById('v-ref').value || 'Virement';
    if (!ben || !amt || amt <= 0) { showToast('❌ Veuillez remplir les champs'); return; }
    if (amt > solde) { showToast('❌ Solde insuffisant'); return; }
    solde -= amt;
    updateSolde();

    renderTx(document.getElementById('tx-list-full'), transactions);

    document.getElementById('v-montant').value = '';
    document.getElementById('v-ref').value = '';
    document.getElementById('v-iban').value = '';
  }

    document.getElementById('e-montant').value = '';
  }
